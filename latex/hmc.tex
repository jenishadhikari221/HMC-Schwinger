\section{Introduction}
\label{sec:intro}
Lattice Quantum Chromodynamics (LQCD) provides a non-perturbative framework for studying strongly interacting gauge 
theories from first principles. By discretizing space-time onto a finite lattice, quantum field theories can be formulated 
in a way that makes them amenable to numerical Monte Carlo simulations. 

As an introduction to LQCD methodology, this project studies the 2D-Schwinger model, 
which is field theory of QED in 1+1 dimensions with gauge group $U(1)$. Although significantly simpler than
4-D QCD, the Schwinger model shares many important qualitative features with non-abelian gauge theories.
In particular, it exhibits confinement, topological sectors, etc. One major advantage of the Schwinger model is its reduced dimensionality. 
In 1+1 dimensions, gauge degrees of freedom are highly constrained by Gauss’ law, simplifying both the analytical structure of the theory and 
its numerical implementation. Despite this simplicity, the model retains essential conceptual ingredients of LQCD, including gauge invariance, 
compact link variables, fermion discretization issues, and the need for importance sampling algorithms such as Hybrid/Hamilton Monte Carlo (HMC).


This report is not an original work, unless mention otherwise concpents and methodology are taken directly from~\cite{gupta1998introduction}.
This report begins with a brief overview of the theoretical background of the field theory, followed by a description of 
its lattice formulation. The simulation methodology, including the implementation of the Hybrid Monte Carlo algorithm, is then presented. 
Finally, the procedures for data analysis and the results obtained from the numerical simulations are discussed. 
The source code of the simulation and the tests of the implemented algorithms can be found in~\cite{jenish_github}.



\section{Theory}
\subsection{QFT}
Let's consider the quantum field theory of QED, the Lagrangian density (later simply Lagrangian) is given as \cref{eqn:qed_full}.

\begin{equation}\label{eqn:qed_full}
    \mathcal{L}\big(\bar{\psi},\psi,A_\mu\big)
    =
    \sum_{f=1}^{N_f}
    \bar{\psi}^{(f)}(x)\Big[\gamma^\mu i \big(\partial_\mu + i g A_\mu(x)\big) - m\Big]\psi^{(f)}(x)
    -\frac{1}{4} F_{\mu\nu}(x)F^{\mu\nu}(x)
\end{equation}
In Path Integral formalism an expectation value is given by
\begin{equation}
    \langle O \rangle = \frac{1}{Z} \int \mathcal{D}A_\mu\mathcal{D}\psi\mathcal{D}\bar{\psi} O[A_\mu,\psi,\bar{\psi}] e^{iS}
\end{equation}
\begin{equation}
    Z = \int \mathcal{D}A_\mu\mathcal{D}\psi\mathcal{D}\bar{\psi} e^{iS}
\end{equation}
After performing a Wick's rotation \cref{eqn:wick_rot} the action becomes \cref{eqn:action_E}\footnote{Note the gamma matrices change after rotation.}.

\begin{align}
    S &= \int d^4x \mathcal {L} \\
    t &\to -i\tau \label{eqn:wick_rot}\\
    Z &= \int \mathcal{D}A_\mu\mathcal{D}\psi\mathcal{D}\bar{\psi} e^{-S_E} \\
    S_E &= \int d^4x  
    \bar{\psi}(x)\Big[\gamma^\mu\big(\partial_\mu + i g A_\mu(x)\big) + m\Big]\psi + \frac{1}{4} F_{\mu\nu}(x)F^{\mu\nu}(x) \label{eqn:action_E}\\
    \gamma_E^i &= i \gamma_M^i
\end{align}
\subsection{Lattice discretization}
Space-time is discretized into a two-dimensional lattice with spacing $a$. 
Gauge fields are represented by compact link variables

\begin{equation}
U_\mu(n) = e^{iA\mu(n)} \in U(1)
\end{equation}
defined on links between lattice sites $n$ and $n + \mu$.
For 2D-Schwinger model the fermion matrix reads as,

\begin{equation}
     M[n',n]^{\alpha\beta} = (m_0+2)\delta^{\alpha\beta}\delta_{n',n} -\frac{1}{2}\sum\limits_{\mu=0}^{1}\{(1-\sigma_\mu)^{\alpha\beta}U_{\mu}(n')\delta_{n'+\mu,n} +
        (1+\sigma_\mu)^{\alpha\beta}U^{\dagger}_{\mu}(n'-\mu)\delta_{n'-\mu,n}  \},
\end{equation}
where $\sigma$ are Pauli sigma matrices. The action can be separated into pure gauge part and fermion part as \cref{eqn:disc_action}. 
In limit $a \to 0$ the continuum field Lagrangian is obtained~\cite{gupta1998introduction}. The fermionic action is implemented with 
anti-periodic(periodic) boundary conditions in temporal(spatial)-direction~\cite{rothe2012lattice}.
\begin{align}\label{eqn:disc_action}
    S_G &= \beta\sum_{n\in \Lambda}\sum_{\mu<\nu} \mathcal{R}Tr(1-P_{\mu\nu}(n))\\
    S_F &= \bar{\psi} M \psi
\end{align}

Consider the path integral, using properties of Grassmann number \cref{eq:grassmann_rules} the fermion field can be integrated out to \cref{eqn:only_gaugePart}. 
In numerical methods we want to sample from the distribution, computing the determinant scales complexity with  $\mathcal{O}(n^3)$ at best and space with
$\mathcal{O}(n^2)$, where $n$ is the volume of the lattice. One can implement $M^\dagger M \chi$ as a vector and solve linear equation
with conjugate gradient (CG)~\cite{lecturenote} to get inverse
as $(M^\dagger M)^{-1} \eta$ with $ \eta = M^\dagger \chi$, where $\chi \sim \mathcal{N}(0,1)$
\begin{align}
    \langle \mathcal{O} \rangle &= \frac{1}{Z} \int \mathcal{D}A_\mu\mathcal{D}\psi\mathcal{D}\bar{\psi} \mathcal{O}[A_\mu,\psi,\bar{\psi}] e^{-S_G - S_F}\\
     &= \frac{1}{Z} \int \mathcal{D}A_\mu \mathcal{O}[A_\mu] \det(M^\dagger M) e^{-S_G} \label{eqn:only_gaugePart}\\
     &= \frac{1}{Z} \int \mathcal{D}A_\mu\mathcal{D}\eta\dagger \mathcal{D} \eta \mathcal{O}[A_\mu]  e^{- \left(S_G + \eta\dagger (M^\dagger M)^{-1} \eta \right)} \label{eqn:inverse_f_M}
\end{align}

\section{Simulation}
To sample from the distribution $e^{-S_E}$ a local Metropolis update is extremely expensive 
due to global nature of fermion matrix and suffers from larger autocorrelation. To generate gauge configurations of desired distribution, 
one uses the Hybrid Monte Carlo algorithm and implement global Metropolis update.

One can check $M^\dagger$ is implemented correctly by $\langle a, Mb \rangle -
\langle M^\dagger a, b \rangle$, which agreed within numerical uncertainty, of $ \mathcal{O}(e^{-13}) $. Thus $M^\dagger$ 
is adjoint matirx of $M$. With the test of $\langle M^\dagger Ma, a \rangle$ the scalar product shows positive real values within numerical error, thus $M^\dagger M$ is positive definite.
The convergence of CG-algorithm within given max iteration provides inverse of $M\dagger M$.

\subsection{Hamiltonian Formulation}
Once a fictitious conjugate momenta $P_\mu(x)$ for each gauge link is addesd $S_E \to H$.
The integrand thus becomes a Gaussian integral, which can be calculated analytically. The addition of the momentum doesn't affect the expectation value due to the same contribution 
in numerator and denominator.
\begin{equation}
H = \sum_{x,\mu} \frac{1}{2} P_\mu(x)^2 
+ S_{\text{eff}}[U].
\end{equation}
The partition function becomes
\begin{equation}
Z = \int \mathcal{D}U\, \mathcal{D}P \;
e^{-H[U,P]}.
\end{equation}

\subsection{Molecular Dynamics}

To propose a new gauge link, the fields are evolved in fictitious simulation time $\tau$
according to Hamilton's equation of motion:
\begin{align}
\frac{dU_\mu(x)}{d\tau} &= i P_\mu(x) U_\mu(x), \\
\frac{dP_\mu(x)}{d\tau} &= 
- \frac{\partial S_{\text{eff}}}{\partial \theta_\mu(x)},
\end{align}

The total force can be split into
\begin{equation}
- \frac{\partial S_{\text{eff}}}{\partial \theta_\mu(x)}= F = F_G + F_F,
\end{equation}
where $F_G$ is the gauge force and $F_F$ is the fermionic force.

\begin{align}
    F_G^\mu = -\beta \Im[U_\mu(n)K_\mu(n)] &   \\
\textrm{with } K_\mu(n) = U_\nu(n+\mu)U_\mu^\dagger(n+\nu)U_\nu^\dagger(n) &+ U_\nu^\dagger(n+\mu-\nu) U_\mu^\dagger(n-\nu)U_\nu(n-\nu)|_{\mu \neq \nu } \\
F_F^\mu = \Im[\chi^\dagger(n)(1-\sigma_\mu)U_\mu(n)\xi(n+\mu) &- \chi^\dagger(n+\mu)(1+ \sigma_\mu)U_\mu^\dagger(n)\xi(n)]\\
\textrm{with } \xi = (M^\dagger M)^{-1}\Phi \textrm{ and } \chi = M \xi
\end{align}

The molecular dynamics equations are solved numerically
using the leapfrog integrator, which is symplectic,
time-reversible, and area-preserving~\cite{lecturenote}. The integrator introduces $\mathcal{O}(\epsilon^2)$ errors.

To test the correctness of the Leapfrog-algorithm the momentum was flipped (i.e. sign changed) after MD and the MD was computed once again. 
The difference between initial and final gauge field agrees within numerical uncertainties $\mathcal{O}(10^{-13})$.

\subsection{Hamiltonian Monte Carlo}

The Hamiltonian Monte Carlo (HMC)~\cref{algorithm:HMC} combines molecular dynamics evolution with the Metropolis accept/reject, which
fulfills detailled balance condition. 
This approach generates statistically independent gauge configurations while efficiently exploring the configuration space.



\begin{algorithm}[H]
    \caption{Hamiltonian Monte Carlo Algorithm}\label{algorithm:HMC}
    \begin{algorithmic}[1]
        \State Initialize $U_\mu(n) \in U(1)$ and $P_\mu(n)\in \mathcal{N}(0, 1) $
        \For{$i = 1$ to $N_{\text{traj}}$}
            \State Generate pseudofermions: $\Phi = M^\dagger \chi\qquad \chi \sim \mathcal{N}(0, 1)$
            \State Sample random momenta: $P_\mu(n) \sim \mathcal{N}(0, 1)$ 
            \State Calculate initial Hamiltonian: $H_{\text{old}} = H[U, P]$
            \State Evolve system for $\tau$ trajectory time using leapfrog integrator 
            \State Calculate final Hamiltonian: $H_{\text{new}} = H[U', P']$
            \State Compute acceptance probability: $\alpha = \min(1, e^{-(H_{\text{new}} - H_{\text{old}})})$
            \If{$\mathcal{R}[0,1] < \alpha$}
                \State Accept new configuration: $U \leftarrow U'$, $P \leftarrow P'$
            \Else
                \State Reject and keep current configuration
            \EndIf
        \EndFor
    \end{algorithmic}
\end{algorithm}
