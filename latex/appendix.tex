\begin{equation}
\int d\psi = 0,
\qquad
\int d\bar{\psi} = 0,
\qquad
\int d\psi \, \psi = i,
\qquad
\int d\bar{\psi} \, \bar{\psi} = i,
\end{equation}

\begin{equation}
\int d\bar{\psi} \, d\psi = 0,
\int d\bar{\psi} \, d\psi \; \psi = 0,
\int d\bar{\psi} \, d\psi \; \bar{\psi} = 0,
\int d\bar{\psi} \, d\psi \; \bar{\psi} \psi = 1.\label{eq:grassmann_rules}
\end{equation}
